Exemplo de Backend com Flask

from flask import Flask, render_template, request
app = Flask(__name__)
tarefas = []
id_counter = 1

def adicionar_tarefa(texto):
    global id_counter
    tarefas.append({
      "id": id_counter,
      "texto": texto,
      "completa": False
    })
    id_counter += 1

@app.route("/")
def index():
    return render_template( "index.html", tarefas=tarefas)

# Aplicação Flask simples para controle de tarefas
app.run()
